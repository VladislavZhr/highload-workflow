BROKER SCHEMA fico.pledged.mortgage.bki

CREATE COMPUTE MODULE NbuHandler_MappingResponse
	CREATE FUNCTION Main() RETURNS BOOLEAN
	BEGIN
		-- Set application number to Environment
		SET Environment.usr.AppNum = InputRoot.MQRFH2.usr.AppNum;
		
		-- Copy main Message
		SET OutputRoot.MQMD = InputRoot.MQMD;
		SET OutputRoot.MQRFH2 = InputRoot.MQRFH2;
		
		SET OutputRoot.XMLNSC = InputRoot.XMLNSC.root.main;
		
		DECLARE counterpartyItem REFERENCE TO InputRoot.XMLNSC.root.counterparties.counterparty[1];
		WHILE LASTMOVE(counterpartyItem) DO
			CALL FillNbcrElement(counterpartyItem);
			MOVE counterpartyItem NEXTSIBLING;
		END WHILE;
		
		RETURN TRUE;
	END;
	
	
	CREATE PROCEDURE SetResultStatusBase(
	  IN outResult REFERENCE,
	  IN counterpartyID CHARACTER,
	  IN DataSourceCode CHARACTER,
	  IN serviceCode CHARACTER,
	  IN requestResultCode CHARACTER)
	BEGIN
	  IF NOT EXISTS(outResult.ESBResultStatuses[]) THEN
		    CREATE LASTCHILD OF outResult NAME 'ESBResultStatuses';
	  END IF;
	  
	  CREATE LASTCHILD OF outResult.ESBResultStatuses NAME 'RequestResultStatus';
	  DECLARE lastRequestResultStatus REFERENCE TO outResult.ESBResultStatuses.RequestResultStatus[<];
	  SET lastRequestResultStatus.(XMLNSC.Attribute)CounterpartyID = counterpartyID;
	  SET lastRequestResultStatus.(XMLNSC.Attribute)DataSourceCode = DataSourceCode;--'NBU'
	  SET lastRequestResultStatus.(XMLNSC.Attribute)ServiceCode = serviceCode;
	  SET lastRequestResultStatus.(XMLNSC.Attribute)RequestResultCode = requestResultCode;
	END;
	

	
	
	CREATE PROCEDURE SetResultStatusErr(IN outResult REFERENCE, 
									   IN counterpartyID CHARACTER,
									   IN DataSourceCode CHARACTER,
									   IN serviceCode CHARACTER,
									   IN requestResultCode CHARACTER,   
									   IN esbSuffix CHARACTER,   
									   IN errorDescription CHARACTER)
	BEGIN
		CALL SetResultStatusBase(outResult, counterpartyID, DataSourceCode, serviceCode, requestResultCode);
		DECLARE lastRequestResultStatus REFERENCE TO outResult.ESBResultStatuses.RequestResultStatus[<];
		SET lastRequestResultStatus.(XMLNSC.Attribute)ErrorMessage     = 'ESB' || Environment.usr.AppNum || esbSuffix;
		SET lastRequestResultStatus.(XMLNSC.Attribute)ErrorDescription = errorDescription;	  
	END;
	
	CREATE PROCEDURE SetResultStatusOk(IN outResult REFERENCE, 
									   IN counterpartyID CHARACTER,
									   IN DataSourceCode CHARACTER,
									   IN serviceCode CHARACTER,
									   IN requestResultCode CHARACTER)
	BEGIN
		CALL SetResultStatusBase(outResult, counterpartyID, DataSourceCode, serviceCode, requestResultCode);
	END;
	

	CREATE PROCEDURE SetAttrIfNotNull (
	  INOUT targetRef REFERENCE,
	  IN    attrName  CHARACTER,
	  IN    valueRef  REFERENCE
	)
	BEGIN
	  IF FIELDNAME(targetRef) IS NOT NULL THEN
	    DECLARE val CHARACTER FIELDVALUE(valueRef);
	    IF val IS NOT NULL AND TRIM(val) <> '' THEN
	      SET targetRef.(XMLNSC.Attribute){attrName} = CAST(val AS CHARACTER CCSID 1208);
	    END IF;
	  END IF;
	END;



	CREATE PROCEDURE FillNbcrElement(IN inputDataRef REFERENCE) BEGIN
		DECLARE counterpartyID CHARACTER CAST(inputDataRef.CounterpartyID AS CHARACTER DEFAULT '');
		DECLARE retryStatus CHARACTER LOWER(inputDataRef.RetryStatus);
		DECLARE isEmptyIpnAndPassport BOOLEAN CAST(inputDataRef.isEmptyIpnAndPassport AS BOOLEAN DEFAULT FALSE);
		DECLARE isEmptyInnForNotRes BOOLEAN CAST(inputDataRef.isEmptyInnForNotRes AS BOOLEAN DEFAULT FALSE); 
		DECLARE isEmptyIsResident BOOLEAN CAST(inputDataRef.isEmptyIsResident AS BOOLEAN DEFAULT FALSE);
		DECLARE UnexpectedPassportDocType BOOLEAN CAST(inputDataRef.UnexpectedPassportDocType AS BOOLEAN DEFAULT FALSE);
		DECLARE isResident BOOLEAN CAST(inputDataRef.IsResident AS BOOLEAN);
		DECLARE serviceCode CHARACTER;
		
		IF isResident then
			SET serviceCode = 'NBCRPPLoanList';
		ELSEIF NOT isResident then
			SET serviceCode = 'NBCRNOTRESLoanList';
		ELSE
			SET serviceCode = '';
		END IF;
		
		
		-- Fill error limit response
		IF retryStatus = 'limit' THEN
		  CALL SetResultStatusErr(OutputRoot.XMLNSC.Application.Bank.Response, counterpartyID, 'NBU', serviceCode, '-100', '200', 'Вичерпано ліміт повторних запитів до NBU');
		  
		-- Fill error response
		ELSEIF retryStatus = 'error' THEN
		  	CALL SetResultStatusErr(OutputRoot.XMLNSC.Application.Bank.Response, counterpartyID, 'NBU', serviceCode, '-100', '200', 'Внутрішня помилка ІШД при взаємодії з NBU');
		  	
		ELSEIF isEmptyIpnAndPassport THEN
			-- ESBResultStatuses
			CALL SetResultStatusErr(OutputRoot.XMLNSC.Application.Bank.Response, counterpartyID, 'NBU', serviceCode, '0' ,'201', 'Запит не був виконаний, відстуній парметри: інн та паспорт');
			
		ELSEIF isEmptyInnForNotRes THEN
			-- ESBResultStatuses
			CALL SetResultStatusErr(OutputRoot.XMLNSC.Application.Bank.Response, counterpartyID, 'NBU', serviceCode, '0' ,'201', 'Запит не був виконаний, відсутній інн для нерезидента');
			
		ELSEIF isEmptyIsResident THEN
			-- ESBResultStatuses
			CALL SetResultStatusErr(OutputRoot.XMLNSC.Application.Bank.Response, counterpartyID, 'NBU', serviceCode, '0' ,'201', 'Запит не був виконаний, відсутній параметр IsResident');
			
		ELSEIF UnexpectedPassportDocType THEN
			-- ESBResultStatuses
			CALL SetResultStatusErr(OutputRoot.XMLNSC.Application.Bank.Response, counterpartyID, 'NBU', serviceCode, '0' ,'201', 'Запит не був виконаний: непідтримуваний тип паспорта');
			
		-- Fill success response
		ELSE
			-- Create Application/Bank/Response/NBCR (PP або PPnotRes)
			IF EXISTS(inputDataRef.getCustomerPP.BKIResponse[]) OR EXISTS(inputDataRef.getCustomerPPnotRes.BKIResponse[]) THEN
			     CALL CreateNbcrElement(counterpartyID);
			END IF;
			
			  
			-- Individual person mapping (204 for PP or PPnotRes)
			IF (EXISTS(inputDataRef.getCustomerPP.ResponseStatus[]) 
				        AND inputDataRef.getCustomerPP.ResponseStatus = '204')
				 OR (EXISTS(inputDataRef.getCustomerPPnotRes.ResponseStatus[]) 
				        AND inputDataRef.getCustomerPPnotRes.ResponseStatus = '204') THEN
			
			  	 CALL SetResultStatusErr(OutputRoot.XMLNSC.Application.Bank.Response, counterpartyID, 'NBU', serviceCode, '2', '204', 'Клієнт не знайдений в БКІ');     

				
			ELSEIF EXISTS(inputDataRef.getCustomerPP.BKIResponse[])
				OR EXISTS(inputDataRef.getCustomerPPnotRes.BKIResponse[]) THEN
				
				DECLARE respParent REFERENCE TO inputDataRef.getCustomerPP;
				DECLARE nonRes REFERENCE TO inputDataRef.getCustomerPPnotRes;
				
				IF NOT EXISTS(respParent.BKIResponse[]) AND EXISTS(nonRes.BKIResponse[]) THEN
				  MOVE respParent TO nonRes;
				END IF;

				
				DECLARE nbcr REFERENCE TO OutputRoot.XMLNSC.Application.Bank.Response.NBCR[<];
				-- Create Application/Bank/Response/NBCR/Individual
				CREATE LASTCHILD OF nbcr NAME 'Individual';
				DECLARE individual REFERENCE TO nbcr.Individual[<];
				
				CREATE LASTCHILD OF individual NAME 'InquiryList';
				SET individual.InquiryList.(XMLNSC.Attribute)DayNumber = 0;
				SET individual.InquiryList.(XMLNSC.Attribute)WeekNumber = 0;
				SET individual.InquiryList.(XMLNSC.Attribute)MonthNumber = 0;
				SET individual.InquiryList.(XMLNSC.Attribute)QuarterNumber = 0;
				SET individual.InquiryList.(XMLNSC.Attribute)YearNumber = 0;
				
				CREATE LASTCHILD OF individual.InquiryList NAME 'Inquiry';
				SET individual.InquiryList.Inquiry.(XMLNSC.Attribute)RequestDate = respParent.BKIResponse.data.kvi_date;
				
				-- Create Application/Bank/Response/NBCR/Individual/Receipts
				CREATE LASTCHILD OF individual NAME 'Receipts';
				DECLARE resultKviItem REFERENCE TO respParent.BKIResponse.data.result_kvi.Item[1];
				WHILE LASTMOVE(resultKviItem) DO
					CREATE LASTCHILD OF individual.Receipts NAME 'Receipt';
					SET individual.Receipts.Receipt[<].(XMLNSC.Attribute)ReceiptNumber = resultKviItem.orderBank;
					SET individual.Receipts.Receipt[<].(XMLNSC.Attribute)MarkOfOwner = resultKviItem.markOfOwner;
					SET individual.Receipts.Receipt[<].(XMLNSC.Attribute)LastName = resultKviItem.fio.lastName;
					SET individual.Receipts.Receipt[<].(XMLNSC.Attribute)FirstName = resultKviItem.fio.firstName;
					SET individual.Receipts.Receipt[<].(XMLNSC.Attribute)MiddleName = resultKviItem.fio.middleName;
					SET individual.Receipts.Receipt[<].(XMLNSC.Attribute)BirthDate = resultKviItem.birthDay;
					SET individual.Receipts.Receipt[<].(XMLNSC.Attribute)TaxNumber = resultKviItem.inn;
					SET individual.Receipts.Receipt[<].(XMLNSC.Attribute)FlagK060 = resultKviItem.flagK060;
					
					-- Create Application/Bank/Response/NBCR/Individual/Receipts/Receipt/LoanContracts
					CREATE LASTCHILD OF individual.Receipts.Receipt[<] NAME 'LoanContracts';
					DECLARE loanContracts REFERENCE TO individual.Receipts.Receipt[<].LoanContracts;
					DECLARE creditsItem REFERENCE TO resultKviItem.credits.Item[1];
					WHILE LASTMOVE(creditsItem) DO
						CREATE LASTCHILD OF loanContracts NAME 'LoanContract';
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)LoanType = creditsItem.typeCredit;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)ContractReference = creditsItem.numberDog;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)BookingDate = creditsItem.dogDay;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)MaturityDate = creditsItem.endDay;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)LimitAmount =  CAST(creditsItem.sumZagal AS DECIMAL) / 100;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)CurrencyCode = creditsItem.r030;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)OutstandingAmount = CAST(creditsItem.sumArrears AS DECIMAL) / 100;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)PrincipalAmount = CAST(creditsItem.sumArrearsBase AS DECIMAL) / 100;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)InterestAmount = CAST(creditsItem.sumArrearsProc AS DECIMAL) / 100;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)PrincipalOverdue = CAST(creditsItem.arrearBase AS DECIMAL) / 100;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)InterestOverdue = CAST(creditsItem.arrearProc AS DECIMAL) / 100;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)OverdueDaysPrincipal = creditsItem.dayBase;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)OverdueDaysInterest = creditsItem.dayProc;
						SET loanContracts.LoanContract[<].(XMLNSC.Attribute)Class = creditsItem.klass;
						
						-- Create Application/Bank/Response/NBCR/Individual/Receipts/Receipt/LoanContracts/LoanContract/TrancheList
						CREATE LASTCHILD OF loanContracts.LoanContract[<] NAME 'TrancheList';
						DECLARE trancheList REFERENCE TO loanContracts.LoanContract[<].TrancheList;
						DECLARE trancheItem REFERENCE TO creditsItem.tranche.Item[1];
						WHILE LASTMOVE(trancheItem) DO
							CREATE LASTCHILD OF trancheList NAME 'Tranche';
							SET trancheList.Tranche[<].(XMLNSC.Attribute)TrancheReference = trancheItem.numDogTr;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)BookingDate = trancheItem.dogDayTr;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)MaturityDate = trancheItem.endDayTr;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)TrancheAmount = CAST(trancheItem.sumZagalTr AS DECIMAL) / 100;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)CurrencyCode = trancheItem.r030Tr;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)PrincipalAmount = CAST(trancheItem.sumArrearsTrBase AS DECIMAL) / 100;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)InterestAmount = CAST(trancheItem.sumArrearsTrProc AS DECIMAL) / 100;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)PrincipalOverdue = CAST(trancheItem.arrearBaseTr AS DECIMAL) / 100;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)InterestOverdue = CAST(trancheItem.arrearProcTr AS DECIMAL) / 100;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)OverdueDaysPrincipal = trancheItem.dayBaseTr;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)OverdueDaysInterest = trancheItem.dayProcTr;
							SET trancheList.Tranche[<].(XMLNSC.Attribute)Class = trancheItem.klassTr;
							
							MOVE trancheItem NEXTSIBLING;
						END WHILE;
						
						MOVE creditsItem NEXTSIBLING;
					END WHILE;
					
					-- Application/Bank/Response/NBCR/Individual/Receipts/Receipt/PledgeList
					CREATE LASTCHILD OF individual.Receipts.Receipt[<] NAME 'PledgeList';
					DECLARE pledgeList REFERENCE TO individual.Receipts.Receipt[<].PledgeList;
					DECLARE pledgeItem REFERENCE TO resultKviItem.pledge.Item[1];
					WHILE LASTMOVE(pledgeItem) DO
						CREATE LASTCHILD OF pledgeList NAME 'Pledge';
						SET pledgeList.Pledge[<].(XMLNSC.Attribute)PledgeDate = pledgeItem.pledgeDay;
						SET pledgeList.Pledge[<].(XMLNSC.Attribute)PledgeType = pledgeItem.s031;
						SET pledgeList.Pledge[<].(XMLNSC.Attribute)PledgeNumber = pledgeItem.orderZastava;
						
						MOVE pledgeItem NEXTSIBLING;
					END WHILE;
					
					MOVE resultKviItem NEXTSIBLING;
				END WHILE;
					
					
				-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor
				CREATE LASTCHILD OF individual NAME 'InfoCreditors';
				DECLARE infoCreditorItem REFERENCE TO respParent.BKIResponse.data.info_creditor.Item[1];
				WHILE LASTMOVE(infoCreditorItem) DO
					CREATE LASTCHILD OF individual.InfoCreditors NAME 'InfoCreditor';
					DECLARE infoCreditor REFERENCE TO individual.InfoCreditors.InfoCreditor[<];
					SET infoCreditor.(XMLNSC.Attribute)Creditor = infoCreditorItem.creditor;
					SET infoCreditor.(XMLNSC.Attribute)CreditorInfo = infoCreditorItem.creditor_info;
					SET infoCreditor.(XMLNSC.Attribute)RelationshipPersonK060 = infoCreditorItem.k060_resp_relation;
					SET infoCreditor.(XMLNSC.Attribute)DebtorsComment = infoCreditorItem.comment;
					
					CREATE LASTCHILD OF infoCreditor NAME 'IdentPerson';
					DECLARE identPerson REFERENCE TO infoCreditor.IdentPerson;
					
					SET infoCreditor.IdentPerson.(XMLNSC.Attribute)LastName = infoCreditorItem.ind_person.last_name;
					SET infoCreditor.IdentPerson.(XMLNSC.Attribute)FirstName = infoCreditorItem.ind_person.first_name;
					SET infoCreditor.IdentPerson.(XMLNSC.Attribute)MiddleName = infoCreditorItem.ind_person.patronymic;
					--SET infoCreditor.IdentPerson.(XMLNSC.Attribute)BirthDate = infoCreditorItem.ind_person.birth_date;
					CALL SetAttrIfNotNull(identPerson, 'BirthDate', infoCreditorItem.ind_person.birth_date);
					SET infoCreditor.IdentPerson.(XMLNSC.Attribute)TaxNumber = infoCreditorItem.ind_person.ind_person_code;
					SET infoCreditor.IdentPerson.(XMLNSC.Attribute)Passport = infoCreditorItem.ind_person.passport;
					SET infoCreditor.IdentPerson.(XMLNSC.Attribute)DocumentNumber = infoCreditorItem.ind_person.document_number;
					
					-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor/Credits/Credit
					CREATE LASTCHILD OF infoCreditor NAME 'Credits';
					DECLARE creditItem REFERENCE TO infoCreditorItem.credit.Item[1];
					WHILE LASTMOVE(creditItem) DO
						CREATE LASTCHILD OF infoCreditor.Credits NAME 'Credit';
						DECLARE credit REFERENCE TO infoCreditor.Credits.Credit[<];
						
						-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor/Credits/Credit/Liability
						CREATE LASTCHILD OF credit NAME 'Liability';
						DECLARE liability REFERENCE TO credit.Liability;
						
					
						SET credit.Liability.(XMLNSC.Attribute)LoanType = creditItem.liability.f037_loan_type;
						SET credit.Liability.(XMLNSC.Attribute)AgreementNumber = creditItem.liability.agreem_no;
						SET credit.Liability.(XMLNSC.Attribute)AgreementStartDate = creditItem.liability.agreem_start_date;
						CALL SetAttrIfNotNull(liability, 'AgreementEndDate', creditItem.liability.obligation_end_date);
						SET credit.Liability.(XMLNSC.Attribute)CurrencyCode = creditItem.liability.r030_currency;
						SET credit.Liability.(XMLNSC.Attribute)TotalAmount = creditItem.liability.total_amount;
						SET credit.Liability.(XMLNSC.Attribute)BalanceAmount = creditItem.liability.balance_off;
						SET credit.Liability.(XMLNSC.Attribute)OverdueAmount = creditItem.liability.overdue_amount;
						
						CALL SetAttrIfNotNull(liability, 'OverdueDays', creditItem.liability.days_overdue);
						
						SET credit.Liability.(XMLNSC.Attribute)CRTypeAssessment = creditItem.liability.s083_risk_type_assessment;
						SET credit.Liability.(XMLNSC.Attribute)Class = creditItem.liability.class;
						SET credit.Liability.(XMLNSC.Attribute)ClassAdjusted = creditItem.liability.korr_class;
						
						-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor/Credits/Credit/Liability/Factors/Factor
						CREATE LASTCHILD OF credit.Liability NAME 'Factors';
						DECLARE liabilityFactorsItem REFERENCE TO creditItem.liability.factors.Item[1];
						WHILE LASTMOVE(liabilityFactorsItem) DO
							CREATE LASTCHILD OF credit.Liability.Factors NAME 'Factor';
							DECLARE factor REFERENCE TO credit.Liability.Factors.Factor[<];
							SET factor.(XMLNSC.Attribute)CRDefaultEvent = liabilityFactorsItem.f075g_risk_event_list;
							SET factor.(XMLNSC.Attribute)CRDefaultEventCode = liabilityFactorsItem.risk_event_code;
							
							MOVE liabilityFactorsItem NEXTSIBLING;
						END WHILE;
						
						-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor/Credits/Credit/Liability/CreditTrancheList/CreditTranche
						CREATE LASTCHILD OF credit.Liability NAME 'CreditTrancheList';
						DECLARE liabilityTrancheItem REFERENCE TO creditItem.liability.tranche.Item[1];
						WHILE LASTMOVE(liabilityTrancheItem) DO
							CREATE LASTCHILD OF credit.Liability.CreditTrancheList NAME 'CreditTranche';
							DECLARE creditTranche REFERENCE TO credit.Liability.CreditTrancheList.CreditTranche[<];
							
							SET creditTranche.(XMLNSC.Attribute)LoanType = liabilityTrancheItem.f037_loan_type;
							SET creditTranche.(XMLNSC.Attribute)AgreementStartDate = liabilityTrancheItem.agreem_start_date;
							CALL SetAttrIfNotNull(creditTranche, 'AgreementEndDate', liabilityTrancheItem.obligation_end_date);
							SET creditTranche.(XMLNSC.Attribute)CurrencyCode = liabilityTrancheItem.r030_currency;
							SET creditTranche.(XMLNSC.Attribute)TotalAmount = liabilityTrancheItem.total_amount;
							SET creditTranche.(XMLNSC.Attribute)BalanceAmount = liabilityTrancheItem.balance_off;
							SET creditTranche.(XMLNSC.Attribute)OverdueAmount = liabilityTrancheItem.overdue_amount;
							CALL SetAttrIfNotNull(creditTranche, 'OverdueDays', liabilityTrancheItem.days_overdue);
							SET creditTranche.(XMLNSC.Attribute)CRTypeAssessment = liabilityTrancheItem.s083_risk_type_assessment;
							SET creditTranche.(XMLNSC.Attribute)Class = liabilityTrancheItem.class;
							SET creditTranche.(XMLNSC.Attribute)ClassAdjusted = liabilityTrancheItem.korr_class;
							
							CREATE LASTCHILD OF creditTranche NAME 'Factors';
							DECLARE liabilityTrancheFactorsItem REFERENCE TO liabilityTrancheItem.factors.Item[1];
							WHILE LASTMOVE(liabilityTrancheFactorsItem) DO
								CREATE LASTCHILD OF creditTranche.Factors NAME 'Factor';
								DECLARE creditTrancheFactor REFERENCE TO creditTranche.Factors.Factor[<];
								SET creditTrancheFactor.(XMLNSC.Attribute)CRDefaultEvent = liabilityTrancheFactorsItem.f075g_risk_event_list;
								SET creditTrancheFactor.(XMLNSC.Attribute)CRDefaultEventCode = liabilityTrancheFactorsItem.risk_event_code;
								
								MOVE liabilityTrancheFactorsItem NEXTSIBLING;
							END WHILE;
							
							MOVE liabilityTrancheItem NEXTSIBLING;
						END WHILE;
						
						-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor/Credits/Credit/Liability/Collaterals/Collateral
						CREATE LASTCHILD OF credit.Liability NAME 'Collaterals';
						DECLARE liabilityCollateralItem REFERENCE TO creditItem.liability.collateral.Item[1];
						WHILE LASTMOVE(liabilityCollateralItem) DO
							CREATE LASTCHILD OF credit.Liability.Collaterals NAME 'Collateral';
							DECLARE collateral REFERENCE TO credit.Liability.Collaterals.Collateral[<];
							SET collateral.(XMLNSC.Attribute)AgreementStartDate = liabilityCollateralItem.agreem_start_date;
							
							CREATE LASTCHILD OF collateral NAME 'Movables';
							DECLARE liabilityCollateralMovableItem REFERENCE TO liabilityCollateralItem.movable.Item[1];
							WHILE LASTMOVE(liabilityCollateralMovableItem) DO
								CREATE LASTCHILD OF collateral.Movables NAME 'CollatList';
								DECLARE movablesCollatList REFERENCE TO collateral.Movables.CollatList[<];
								SET movablesCollatList.(XMLNSC.Attribute)CollateralType = liabilityCollateralMovableItem.s031_col_type;
								SET movablesCollatList.(XMLNSC.Attribute)CurrencyCode = liabilityCollateralMovableItem.r030_currency;
								SET movablesCollatList.(XMLNSC.Attribute)CollateralAmount = liabilityCollateralMovableItem.collateral_amount;
								
								MOVE liabilityCollateralMovableItem NEXTSIBLING;
							END WHILE;
							
							CREATE LASTCHILD OF collateral NAME 'Immovables';
							DECLARE liabilityCollateralImmovableItem REFERENCE TO liabilityCollateralItem.immovable.Item[1];
							WHILE LASTMOVE(liabilityCollateralImmovableItem) DO
								CREATE LASTCHILD OF collateral.Immovables NAME 'CollatList';
								DECLARE immovablesCollatList REFERENCE TO collateral.Immovables.CollatList[<];
								SET immovablesCollatList.(XMLNSC.Attribute)CollateralType = liabilityCollateralImmovableItem.s031_col_type;
								SET immovablesCollatList.(XMLNSC.Attribute)CurrencyCode = liabilityCollateralImmovableItem.r030_currency;
								SET immovablesCollatList.(XMLNSC.Attribute)CollateralAmount = liabilityCollateralImmovableItem.collateral_amount;
								
								MOVE liabilityCollateralImmovableItem NEXTSIBLING;
							END WHILE;
							
							CREATE LASTCHILD OF collateral NAME 'CollatDeposits';
							DECLARE liabilityCollateralDepositItem REFERENCE TO liabilityCollateralItem.deposit.Item[1];
							WHILE LASTMOVE(liabilityCollateralDepositItem) DO
								CREATE LASTCHILD OF collateral.CollatDeposits NAME 'CollatList';
								DECLARE collatDepositsCollatList REFERENCE TO collateral.CollatDeposits.CollatList[<];
								SET collatDepositsCollatList.(XMLNSC.Attribute)CurrencyCode = liabilityCollateralDepositItem.r030_currency;
								SET collatDepositsCollatList.(XMLNSC.Attribute)CollateralAmount = liabilityCollateralDepositItem.collateral_amount;
								
								MOVE liabilityCollateralDepositItem NEXTSIBLING;
							END WHILE;
							
							MOVE liabilityCollateralItem NEXTSIBLING;
						END WHILE;
						
						
						
						
						-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor/Credits/Credit/CreditLoans/CreditLoan
						CREATE LASTCHILD OF credit NAME 'CreditLoans';
						DECLARE loanItem REFERENCE TO creditItem.loan.Item[1];
						WHILE LASTMOVE(loanItem) DO
							CREATE LASTCHILD OF credit.CreditLoans NAME 'CreditLoan';
							DECLARE creditLoans REFERENCE TO credit.CreditLoans.CreditLoan[<];
							
							SET creditLoans.(XMLNSC.Attribute)LoanType = loanItem.f037_loan_type;
							SET creditLoans.(XMLNSC.Attribute)AgreementNumber = loanItem.agreem_no;
							CALL SetAttrIfNotNull(creditLoans, 'AgreementStartDate', loanItem.agreem_start_date);
							CALL SetAttrIfNotNull(creditLoans, 'AgreementEndDate', loanItem.agreem_end_date);
							SET creditLoans.(XMLNSC.Attribute)CurrencyCode = loanItem.r030_currency;
							SET creditLoans.(XMLNSC.Attribute)TotalAmount = loanItem.total_amount;
							SET creditLoans.(XMLNSC.Attribute)BalanceAmount = loanItem.balance_on;
							SET creditLoans.(XMLNSC.Attribute)OverdueAmount = loanItem.overdue_amount;
							SET creditLoans.(XMLNSC.Attribute)OverdueDays = loanItem.days_overdue;
							
							SET creditLoans.(XMLNSC.Attribute)DebtReservesWritten = loanItem.debt_wr_on;
							SET creditLoans.(XMLNSC.Attribute)IncomeReservesWritten = loanItem.inc_wr_on;
							CALL SetAttrIfNotNull(creditLoans, 'OffBalanceWritten',  loanItem.write_off_date);
							SET creditLoans.(XMLNSC.Attribute)CRTypeAssessment = loanItem.s083_risk_type_assessment;
							
							SET creditLoans.(XMLNSC.Attribute)Class = loanItem.class;
							SET creditLoans.(XMLNSC.Attribute)ClassAdjusted = loanItem.korr_class;
							SET creditLoans.(XMLNSC.Attribute)StateLoanProgram = loanItem.d180_state_loan_prog;
							
							-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor/Credits/CreditLoan/Factors/Factor
							CREATE LASTCHILD OF creditLoans NAME 'Factors';
							DECLARE factorsLoanItem REFERENCE TO loanItem.factors.Item[1];
							WHILE LASTMOVE(factorsLoanItem) DO
								CREATE LASTCHILD OF creditLoans.Factors NAME 'Factor';
								DECLARE factorCrLoan REFERENCE TO creditLoans.Factors.Factor[<];
								SET factorCrLoan.(XMLNSC.Attribute)CRDefaultEvent = factorsLoanItem.f075g_risk_event_list;
								SET factorCrLoan.(XMLNSC.Attribute)CRDefaultEventCode = factorsLoanItem.risk_event_code;
								
								MOVE factorsLoanItem NEXTSIBLING;
							END WHILE;
							
							-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor/Credits/CreditLoan/CreditTrancheList/CreditTranche
							CREATE LASTCHILD OF creditLoans NAME 'CreditTrancheList';
							DECLARE loanTrancheItem REFERENCE TO loanItem.tranche.Item[1];
							WHILE LASTMOVE(loanTrancheItem) DO
								CREATE LASTCHILD OF creditLoans.CreditTrancheList NAME 'CreditTranche';
								DECLARE creditTrancheCrLoan REFERENCE TO creditLoans.CreditTrancheList.CreditTranche[<];
								
								SET creditTrancheCrLoan.(XMLNSC.Attribute)LoanType = loanTrancheItem.f037_loan_type;
								SET creditTrancheCrLoan.(XMLNSC.Attribute)AgreementStartDate = loanTrancheItem.agreem_start_date;
								CALL SetAttrIfNotNull(creditTrancheCrLoan, 'AgreementEndDate', loanTrancheItem.agreem_end_date);
								SET creditTrancheCrLoan.(XMLNSC.Attribute)CurrencyCode = loanTrancheItem.r030_currency;
								SET creditTrancheCrLoan.(XMLNSC.Attribute)TotalAmount = loanTrancheItem.total_amount;
								SET creditTrancheCrLoan.(XMLNSC.Attribute)BalanceAmount = loanTrancheItem.balance_on;
								SET creditTrancheCrLoan.(XMLNSC.Attribute)OverdueAmount = loanTrancheItem.overdue_amount;
								SET creditTrancheCrLoan.(XMLNSC.Attribute)OverdueDays = loanTrancheItem.days_overdue;
								SET creditLoans.(XMLNSC.Attribute)DebtReservesWritten = loanItem.debt_wr_on;
								SET creditLoans.(XMLNSC.Attribute)IncomeReservesWritten = loanItem.inc_wr_on;
								CALL SetAttrIfNotNull(creditLoans, 'OffBalanceWritten', loanItem.write_off_date);
								SET creditTrancheCrLoan.(XMLNSC.Attribute)CRTypeAssessment = loanTrancheItem.s083_risk_type_assessment;
								SET creditTrancheCrLoan.(XMLNSC.Attribute)Class = loanTrancheItem.class;
								SET creditTrancheCrLoan.(XMLNSC.Attribute)ClassAdjusted = loanTrancheItem.korr_class;
								
								CREATE LASTCHILD OF creditTrancheCrLoan NAME 'Factors';
								DECLARE loanTrancheFactorsItem REFERENCE TO loanTrancheItem.factors.Item[1];
								WHILE LASTMOVE(loanTrancheFactorsItem) DO
									CREATE LASTCHILD OF creditTrancheCrLoan.Factors NAME 'Factor';
									DECLARE creditTrancheFactorCrLoan REFERENCE TO creditTrancheCrLoan.Factors.Factor[<];
									SET creditTrancheFactorCrLoan.(XMLNSC.Attribute)CRDefaultEvent = loanTrancheFactorsItem.f075g_risk_event_list;
									SET creditTrancheFactorCrLoan.(XMLNSC.Attribute)CRDefaultEventCode = loanTrancheFactorsItem.risk_event_code;
									
									MOVE loanTrancheFactorsItem NEXTSIBLING;
								END WHILE;
								
								MOVE loanTrancheItem NEXTSIBLING;
							END WHILE;
							
							-- Application/Bank/Response/NBCR/Individual/InfoCreditors/InfoCreditor/Credits/CreditLoan/Collaterals/Collateral
							CREATE LASTCHILD OF creditLoans NAME 'Collaterals';
							DECLARE loanCollateralItem REFERENCE TO loanItem.collateral.Item[1];
							WHILE LASTMOVE(loanCollateralItem) DO
								CREATE LASTCHILD OF creditLoans.Collaterals NAME 'Collateral';
								DECLARE collateralCrLoan REFERENCE TO creditLoans.Collaterals.Collateral[<];
								SET collateralCrLoan.(XMLNSC.Attribute)AgreementStartDate = loanCollateralItem.agreem_start_date;
								
								CREATE LASTCHILD OF collateralCrLoan NAME 'Movables';
								DECLARE loanCollateralMovableItem REFERENCE TO loanCollateralItem.movable.Item[1];
								WHILE LASTMOVE(loanCollateralMovableItem) DO
									CREATE LASTCHILD OF collateralCrLoan.Movables NAME 'CollatList';
									DECLARE movablesCollatListCrLoan REFERENCE TO collateralCrLoan.Movables.CollatList[<];
									SET movablesCollatListCrLoan.(XMLNSC.Attribute)CollateralType = loanCollateralMovableItem.s031_col_type;
									SET movablesCollatListCrLoan.(XMLNSC.Attribute)CurrencyCode = loanCollateralMovableItem.r030_currency;
									SET movablesCollatListCrLoan.(XMLNSC.Attribute)CollateralAmount = loanCollateralMovableItem.collateral_amount;
									
									MOVE loanCollateralMovableItem NEXTSIBLING;
								END WHILE;
								
								CREATE LASTCHILD OF collateralCrLoan NAME 'Immovables';
								DECLARE loanCollateralImmovableItem REFERENCE TO loanCollateralItem.immovable.Item[1];
								WHILE LASTMOVE(loanCollateralImmovableItem) DO
									CREATE LASTCHILD OF collateralCrLoan.Immovables NAME 'CollatList';
									DECLARE immovablesCollatListCrLoan REFERENCE TO collateralCrLoan.Immovables.CollatList[<];
									SET immovablesCollatListCrLoan.(XMLNSC.Attribute)CollateralType = loanCollateralImmovableItem.s031_col_type;
									SET immovablesCollatListCrLoan.(XMLNSC.Attribute)CurrencyCode = loanCollateralImmovableItem.r030_currency;
									SET immovablesCollatListCrLoan.(XMLNSC.Attribute)CollateralAmount = loanCollateralImmovableItem.collateral_amount;
									
									MOVE loanCollateralImmovableItem NEXTSIBLING;
								END WHILE;
								
								CREATE LASTCHILD OF collateralCrLoan NAME 'CollatDeposits';
								DECLARE loanCollateralDepositItem REFERENCE TO loanCollateralItem.deposit.Item[1];
								WHILE LASTMOVE(loanCollateralDepositItem) DO
									CREATE LASTCHILD OF collateralCrLoan.CollatDeposits NAME 'CollatList';
									DECLARE collatDepositsCollatListCrLoan REFERENCE TO collateralCrLoan.CollatDeposits.CollatList[<];
									SET collatDepositsCollatListCrLoan.(XMLNSC.Attribute)CurrencyCode = loanCollateralDepositItem.r030_currency;
									SET collatDepositsCollatListCrLoan.(XMLNSC.Attribute)CollateralAmount = loanCollateralDepositItem.collateral_amount;
									
									MOVE loanCollateralDepositItem NEXTSIBLING;
								END WHILE;
								
								MOVE loanCollateralItem NEXTSIBLING;
							END WHILE;
							
							MOVE loanItem NEXTSIBLING;
						END WHILE;
						
						MOVE creditItem NEXTSIBLING;
					END WHILE;
					
					MOVE infoCreditorItem NEXTSIBLING;
				END WHILE;
				
				
				-- Set Application/Bank/Response/ESBResultStatuses/RequestResultStatus
			  	CALL SetResultStatusOK(OutputRoot.XMLNSC.Application.Bank.Response, counterpartyID, 'NBU',serviceCode,'1'); 		
			END IF;
		END IF;
	END;
	
	
	CREATE PROCEDURE CreateNbcrElement(IN counterpartyID CHARACTER) BEGIN
		-- Create Application/Bank/Response/NBCR
		IF EXISTS(OutputRoot.XMLNSC.Application.Bank.Response.NBCR[]) THEN
			CREATE NEXTSIBLING OF OutputRoot.XMLNSC.Application.Bank.Response.NBCR[<] NAME 'NBCR';
		ELSEIF EXISTS(OutputRoot.XMLNSC.Application.Bank.Response.CheckResult[]) THEN
			CREATE NEXTSIBLING OF OutputRoot.XMLNSC.Application.Bank.Response.CheckResult[<] NAME 'NBCR';
		ELSEIF EXISTS(OutputRoot.XMLNSC.Application.Bank.Response.InternalInformation[]) THEN
			CREATE NEXTSIBLING OF OutputRoot.XMLNSC.Application.Bank.Response.InternalInformation[<] NAME 'NBCR';
		ELSE
			CREATE FIRSTCHILD OF OutputRoot.XMLNSC.Application.Bank.Response NAME 'NBCR';
		END IF;
		
		-- Application/Bank/Response/NBCR/@CounterpartyID
		SET OutputRoot.XMLNSC.Application.Bank.Response.NBCR[<].(XMLNSC.Attribute)CounterpartyID = counterpartyID;
	END;
END MODULE;
